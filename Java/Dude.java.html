import processing.core.PImage;

import java.util.List;

public abstract class Dude extends transform_entity {
    public static final String DUDE_KEY = "dude";
    public static final int DUDE_ACTION_PERIOD_IDX = 0;
    public static final int DUDE_ANIMATION_PERIOD_IDX = 1;
    public static final int DUDE_RESOURCE_LIMIT_IDX = 2;
    public static final int DUDE_NUM_PROPERTIES = 3;
    public Dude(String id, Point position, List<PImage> images, int imageIndex) {
        super(id, position, images, imageIndex);
    }
    public Point nextPositionDude(WorldModel world, Point destPos){
        return nextPositionDude(world, destPos, new SingleStepPathingStrategy());
    }

    public Point nextPositionDude(WorldModel world, Point destPos, PathingStrategy strategy) {

        List<Point> path = strategy.computePath( this.position,destPos, pt-> world.withinBounds(pt) && (!world.isOccupied(pt) || (world.getOccupancyCell(pt) instanceof Stump)),
        (pt1,pt2) -> pt1.adjacent(pt2),
        PathingStrategy.CARDINAL_NEIGHBORS);

        return path.isEmpty() ? this.position : path.get(0);
    }
    }
