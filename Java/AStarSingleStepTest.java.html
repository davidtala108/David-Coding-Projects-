import org.junit.jupiter.api.Test;
import java.util.List;
import java.util.function.Function;
import java.util.function.Predicate;
import java.util.function.BiPredicate;
import java.util.stream.Stream;

import static org.junit.jupiter.api.Assertions.*;

public class AStarSingleStepTest {

    @Test
    public void testSimpleHorizontalPath() {
        AStarPathingStrategy strategy = new AStarPathingStrategy();
        Point start = new Point(0, 0);
        Point goal = new Point(3, 0);

        Predicate<Point> canPassThrough = p -> true;  // all points passable
        BiPredicate<Point, Point> withinReach = (p1, p2) -> p1.adjacent(p2);

        Function<Point, Stream<Point>> neighbors = p -> Stream.of(
                new Point(p.x + 1, p.y),
                new Point(p.x - 1, p.y),
                new Point(p.x, p.y + 1),
                new Point(p.x, p.y - 1)
        );

        List<Point> path = strategy.computePath(start, goal, canPassThrough, withinReach, neighbors);

        // Only one step expected, as strategy returns next move only
        assertEquals(1, path.size());
        assertEquals(new Point(1, 0), path.get(0));
    }

    @Test
    public void testBlockedPath() {
        AStarPathingStrategy strategy = new AStarPathingStrategy();
        Point start = new Point(0, 0);
        Point goal = new Point(1, 0);

        Predicate<Point> canPassThrough = p -> false; // all points are blocked
        BiPredicate<Point, Point> withinReach = (p1, p2) -> p1.adjacent(p2);
        Function<Point, Stream<Point>> neighbors = PathingStrategy.CARDINAL_NEIGHBORS;

        List<Point> path = strategy.computePath(start, goal, canPassThrough, withinReach, neighbors);

        // No path should be found
        assertTrue(path.isEmpty());
    }

    @Test
    public void testAlreadyAtGoal() {
        AStarPathingStrategy strategy = new AStarPathingStrategy();
        Point start = new Point(2, 2);
        Point goal = new Point(2, 2); // already there

        Predicate<Point> canPassThrough = p -> true;
        BiPredicate<Point, Point> withinReach = (p1, p2) -> p1.adjacent(p2);
        Function<Point, Stream<Point>> neighbors = PathingStrategy.CARDINAL_NEIGHBORS;

        List<Point> path = strategy.computePath(start, goal, canPassThrough, withinReach, neighbors);

        // Already at goal, no steps needed
        assertTrue(path.isEmpty());
    }
    private final Function<Point, Stream<Point>> cardinalNeighbors = p -> Stream.of(
            new Point(p.x, p.y - 1), // up
            new Point(p.x, p.y + 1), // down
            new Point(p.x - 1, p.y), // left
            new Point(p.x + 1, p.y)  // right
    );

    @Test
    public void testSimpleStepRight() {
        SingleStepPathingStrategy strategy = new SingleStepPathingStrategy();
        Point start = new Point(1, 1);
        Point goal = new Point(4, 1);  // Rightward

        Predicate<Point> canPassThrough = p -> true;
        BiPredicate<Point, Point> withinReach = Point::adjacent;

        List<Point> path = strategy.computePath(start, goal, canPassThrough, withinReach, cardinalNeighbors);

        assertEquals(1, path.size());
        assertEquals(new Point(2, 1), path.get(0));  // One step right
    }

    @Test
    public void testBlockedRightStep() {
        SingleStepPathingStrategy strategy = new SingleStepPathingStrategy();
        Point start = new Point(1, 1);
        Point goal = new Point(4, 1);

        Predicate<Point> canPassThrough = p -> !p.equals(new Point(2, 1)); // block right neighbor
        BiPredicate<Point, Point> withinReach = Point::adjacent;

        List<Point> path = strategy.computePath(start, goal, canPassThrough, withinReach, cardinalNeighbors);

        // No valid step toward goal
        assertTrue(path.isEmpty());
    }

    @Test
    public void testDownIsCloserThanUp() {
        SingleStepPathingStrategy strategy = new SingleStepPathingStrategy();
        Point start = new Point(2, 2);
        Point goal = new Point(2, 4);  // Down

        Predicate<Point> canPassThrough = p -> true;
        BiPredicate<Point, Point> withinReach = Point::adjacent;

        List<Point> path = strategy.computePath(start, goal, canPassThrough, withinReach, cardinalNeighbors);

        assertEquals(1, path.size());
        assertEquals(new Point(2, 3), path.get(0));  // One step down
    }

    @Test
    public void testNoValidStep() {
        SingleStepPathingStrategy strategy = new SingleStepPathingStrategy();
        Point start = new Point(1, 1);
        Point goal = new Point(1, 0);

        Predicate<Point> canPassThrough = p -> false;  // All neighbors are blocked
        BiPredicate<Point, Point> withinReach = Point::adjacent;

        List<Point> path = strategy.computePath(start, goal, canPassThrough, withinReach, cardinalNeighbors);

        assertTrue(path.isEmpty());
    }

    @Test
    public void testIgnoresEndPoint() {
        SingleStepPathingStrategy strategy = new SingleStepPathingStrategy();
        Point start = new Point(1, 1);
        Point goal = new Point(1, 0);  // Up is the goal

        Predicate<Point> canPassThrough = p -> true;
        BiPredicate<Point, Point> withinReach = Point::adjacent;

        List<Point> path = strategy.computePath(start, goal, canPassThrough, withinReach, cardinalNeighbors);

        // It should ignore the goal point, so no path returned
        assertTrue(path.isEmpty());
    }
}
