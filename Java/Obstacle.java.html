import processing.core.PImage;

import java.util.List;

public class Obstacle extends animate_enity {
    double animationPeriod;

    public Obstacle(String Id, Point point, List<PImage> images, int imageIndex, double animationPeriod) {
        super( Id, point, images, imageIndex);
        this.animationPeriod = animationPeriod;
    }


    public double getAnimationPeriod() {
        return animationPeriod;
    }


    @Override
    public void executeAnimation(EventScheduler scheduler, int repeat) {
        this.nextImage();
        if (repeat != 1) {
            scheduler.scheduleEvent(
                    this,
                    new Animation(this, Math.max(repeat - 1, 0)),
                    this.getAnimationPeriod());
        }
    }
    @Override
    public void scheduleActions(EventScheduler scheduler, WorldModel world, ImageStore imageStore) {
        scheduler.scheduleEvent(this, new Animation(this, 0), this.animationPeriod);
    }

}
