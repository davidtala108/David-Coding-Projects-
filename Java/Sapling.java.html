import processing.core.PImage;

import java.util.List;

public class Sapling extends Plant {
    private final double actionPeriod;
    private final double animationPeriod = 1.000;

    public Sapling(String id, Point position, List<PImage> images, double actionPeriod, int health) {
        super(id, position, images, 0,health);
        this.actionPeriod = actionPeriod;
    }


    public double getAnimationPeriod() {
        return animationPeriod;
    }

    @Override
    public boolean transform(WorldModel world, EventScheduler scheduler, ImageStore imageStore) {
        if (this.getHealth() <= 0) {
            Stump stump = new Stump("stump_" + this.getId(), this.getPosition(), imageStore.getImageList("stump"),0);
            this.setId("stump_" + this.id);
            world.removeEntity(scheduler, this);

            world.addEntity(stump);
            return true;
        }

        if (this.getHealth() >= 5) {
            Tree tree = new Tree("tree_" + this.getId(), this.getPosition(),  imageStore.getImageList("tree"),
                    0,  Functions.getNumFromRange(1.4, 1.0),Functions.getNumFromRange(0.6, 0.05), Functions.getIntFromRange(3, 1));
            this.setId("tree_" + this.id);
            world.removeEntity(scheduler, this);

            world.addEntity(tree);
            tree.scheduleActions(scheduler, world, imageStore);
            return true;
        }

        return false;
    }
    @Override
    public void executeActivity(WorldModel world, ImageStore imageStore, EventScheduler scheduler) {
        this.setHealth(getHealth() + 1);
        if (!this.transformPlant(world, scheduler, imageStore)) {
            scheduler.scheduleEvent(this, new Activity(this, world, imageStore), this.actionPeriod);
        }
    }
    @Override
    public void executeAnimation(EventScheduler scheduler, int repeat) {
        this.nextImage();
        if (repeat != 1) {
            scheduler.scheduleEvent(
                    this,
                    new Animation(this, Math.max(repeat - 1, 0)),
                    this.getAnimationPeriod()
            );
        }
    }
    @Override
    public void scheduleActions(EventScheduler scheduler, WorldModel world, ImageStore imageStore) {
        scheduler.scheduleEvent(this, new Activity(this, world, imageStore), this.actionPeriod);
        scheduler.scheduleEvent(this, new Animation(this, 0), this.animationPeriod);
    }
}
